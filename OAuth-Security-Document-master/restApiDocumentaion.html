/register POST None {"Username":"",emailId:"",Password":""} None Code: 200 Contents: { auth: true, token: token } Code: 500 Content: { error : "There was a problem registering the user." } $.ajax({ url: "/register", data:{"Username":"",emailId:"",Password":""}, dataType: "json", type : "POST", success : function(r) { console.log(r); } }) Registering the User. So a token will be sent to the user in the response.
/login POST None {"Username":"",Password":""} {token:""} Code: 200 Contents Code: 401 Content: { error : "Username or password incorrect" } $.ajax({ url: "/login", data:{Username":"","Password":""}, dataType: "json", type : "POST", success : function(r) { console.log(r); } }) While the login button is clicked.The request goes to the server and the token in the header will be verified
/contents GET None NONE None Code: 200 Content: {CourseName:"",CourseId:"",TopicNames:"", TopicIds:"", "contentdurations:""} Code: 204 Content: { error : "No content available" } $.ajax({ url: "/contents", dataType: "json", type : "GET", success : function(r) { console.log(r); } }) Getting the contents on the landing page.
/course/topic POST {Courseid:"",CourseName:"",topicid:"",topicname:""} NONE None Code: 200 Content:{authorname:"", authordescription"",vidioname:"", vidioPath:""} Code: 204 Content: { error : "No topic available" } $.ajax({ url: "//Course/topic", dataType: "json", type : "POST", success : function(r) { console.log(r); } }) When the particular topic from particular Course will be clicked the vidios corresponding to that Course and topic will be returned
